1%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  2% FloodFill.m unwraps the phase image, avoiding all branch cuts.
  3% Created by B.S. Spottiswoode on 12/10/2004
  4% Last modified on 13/10/2004
  5%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
  6
  7function [IM_unwrapped, rowref, colref] = FloodFill(IM_phase, branch_cuts)
  8
  9[height, width]=size(IM_phase);
 10% figure; imagesc(immultiply(IM_phase, ~branch_cuts)), colormap(gray), axis square, axis off, title('Phase image');
 11% uiwait(msgbox('Select a known true phase reference point','Phase reference point','modal'));
 12% [xref,yref] = ginput(1); 
 13
 14% Determine the positions of the non-branch cut pixels
 15non_branch_cuts = find(branch_cuts < 1);
 16
 17% Determine the row and column indices of these pixels
 18% [y_ind, x_ind] = ind2sub([height, width], non_branch_cuts);
 19y_ind = rem(non_branch_cuts - 1, height) + 1;
 20x_ind = (non_branch_cuts - y_ind) / height + 1;
 21
 22% Determine the center of the array
 23xc = width/2;
 24yc = height/2;
 25
 26% Determine radial position of those points
 27r = sqrt((x_ind - xc).^2 + (y_ind - yc).^2);
 28
 29% This picks the unwrapping starting position
 30% At the non-branch cut pixel closest to the 
 31% center of the spectrum.
 32rowref = y_ind(r == min(r));
 33colref = x_ind(r == min(r));
 34
 35% Flag specifying that the starting point corresponds
 36% to a branch cut
 37% branch_cut_start = branch_cuts(rowref, colref);
 38
 39% if logical(branch_cut_start)
 40%     error('Selected point corresponds to a branch cut.');
 41% end
 42
 43IM_unwrapped=zeros(height,width);
 44unwrapped_binary=zeros(height,width);
 45adjoin=zeros(height,width);
 46
 47adjoin(rowref-1, colref)=1;                                 %Label the first four adjoining pixels
 48adjoin(rowref+1, colref)=1;
 49adjoin(rowref, colref-1)=1;
 50adjoin(rowref, colref+1)=1;
 51IM_unwrapped(rowref, colref)=IM_phase(rowref, colref);      %Mark the first pixel as unwrapped
 52unwrapped_binary(rowref, colref)=1;
 53
 54% disp('Performing floodfill operation ...');
 55count_limit=0;
 56adjoin_stuck=0;
 57while sum(sum(adjoin(2:height-1,2:width-1)))~=0              %Loop until there are no adjoining pixels or they all lie on the border
 58    while count_limit<100                                   %or the code gets stuck because of isolated regions
 59        [r_adjoin, c_adjoin]=find(adjoin);                          %Obtain coordinates of adjoining unwrapped phase pixels
 60        if (length(r_adjoin)==adjoin_stuck)
 61            count_limit=count_limit+1;                              %Make sure loop doesn't get stuck
 62        else
 63            count_limit=0;
 64        end
 65        temp = length(r_adjoin);
 66        adjoin_stuck = temp;
 67        for i=1:temp(1)
 68            r_active=r_adjoin(i);
 69            c_active=c_adjoin(i);
 70            if r_active <= height-1 && r_active>=2 && c_active<=width-1 && c_active>=2                    %Ignore pixels near the border
 71                %First search below for an adjoining unwrapped phase pixel
 72                if branch_cuts(r_active+1, c_active)==0 && unwrapped_binary(r_active+1, c_active)==1
 73                    phase_ref=IM_unwrapped(r_active+1, c_active);                                   %Obtain the reference unwrapped phase
 74                    p=unwrap([phase_ref IM_phase(r_active, c_active)]);                             %Unwrap the active pixel
 75                    IM_unwrapped(r_active, c_active)=p(2);
 76                    unwrapped_binary(r_active, c_active)=1;                                         %Mark the pixel as unwrapped
 77                    adjoin(r_active, c_active)=0;                                                   %Remove it from the list of adjoining pixels
 78                    if r_active-1>=1 && unwrapped_binary(r_active-1, c_active)==0 && branch_cuts(r_active-1, c_active)==0
 79                        adjoin(r_active-1, c_active)=1;
 80                    end
 81                    if c_active-1>=1 && unwrapped_binary(r_active, c_active-1)==0 && branch_cuts(r_active, c_active-1)==0
 82                        adjoin(r_active, c_active-1)=1;
 83                    end
 84                    if c_active+1<=width && unwrapped_binary(r_active, c_active+1)==0 && branch_cuts(r_active, c_active+1)==0
 85                        adjoin(r_active, c_active+1)=1;
 86                    end
 87                end
 88                %Then search above
 89                if branch_cuts(r_active-1, c_active)==0 && unwrapped_binary(r_active-1, c_active)==1
 90                    phase_ref=IM_unwrapped(r_active-1, c_active);                                   %Obtain the reference unwrapped phase
 91                    p=unwrap([phase_ref IM_phase(r_active, c_active)]);                             %Unwrap the active pixel
 92                    IM_unwrapped(r_active, c_active)=p(2);
 93                    unwrapped_binary(r_active, c_active)=1;                                         %Mark the pixel as unwrapped
 94                    adjoin(r_active, c_active)=0;                                                   %Remove it from the list of adjoining pixels
 95                    %Update the new adjoining pixels:
 96                    if r_active+1<=height && unwrapped_binary(r_active+1, c_active)==0 && branch_cuts(r_active+1, c_active)==0
 97                        adjoin(r_active+1, c_active)=1;
 98                    end
 99                    if c_active-1>=1 && unwrapped_binary(r_active, c_active-1)==0 && branch_cuts(r_active, c_active-1)==0
100                        adjoin(r_active, c_active-1)=1;
101                    end
102                    if c_active+1<=width && unwrapped_binary(r_active, c_active+1)==0 && branch_cuts(r_active, c_active+1)==0
103                        adjoin(r_active, c_active+1)=1;
104                    end
105                end
106                %Then search on the right
107                if branch_cuts(r_active, c_active+1)==0 && unwrapped_binary(r_active, c_active+1)==1
108                    phase_ref=IM_unwrapped(r_active, c_active+1);                                   %Obtain the reference unwrapped phase
109                    p=unwrap([phase_ref IM_phase(r_active, c_active)]);                             %Unwrap the active pixel
110                    IM_unwrapped(r_active, c_active)=p(2);
111                    unwrapped_binary(r_active, c_active)=1;                                         %Mark the pixel as unwrapped
112                    adjoin(r_active, c_active)=0;                                                   %Remove it from the list of adjoining pixels
113                    if r_active+1<=height && unwrapped_binary(r_active+1, c_active)==0 && branch_cuts(r_active+1, c_active)==0
114                        adjoin(r_active+1, c_active)=1;
115                    end
116                    if c_active-1>=1 && unwrapped_binary(r_active, c_active-1)==0 && branch_cuts(r_active, c_active-1)==0
117                        adjoin(r_active, c_active-1)=1;
118                    end
119                    if r_active-1>=1 && unwrapped_binary(r_active-1, c_active)==0 && branch_cuts(r_active-1, c_active)==0
120                        adjoin(r_active-1, c_active)=1;
121                    end
122                end
123                %Finally search on the left
124                if branch_cuts(r_active, c_active-1)==0 && unwrapped_binary(r_active, c_active-1)==1
125                    phase_ref=IM_unwrapped(r_active, c_active-1);                                   %Obtain the reference unwrapped phase
126                    p=unwrap([phase_ref IM_phase(r_active, c_active)]);                             %Unwrap the active pixel
127                    IM_unwrapped(r_active, c_active)=p(2);
128                    unwrapped_binary(r_active, c_active)=1;                                         %Mark the pixel as unwrapped
129                    adjoin(r_active, c_active)=0;                                                   %Remove it from the list of adjoining pixels
130                    if r_active+1<=height && unwrapped_binary(r_active+1, c_active)==0 && branch_cuts(r_active+1, c_active)==0
131                        adjoin(r_active+1, c_active)=1;
132                    end
133                    if c_active+1<=width && unwrapped_binary(r_active, c_active+1)==0 && branch_cuts(r_active, c_active+1)==0
134                        adjoin(r_active, c_active+1)=1;
135                    end
136                    if r_active-1>=1 && unwrapped_binary(r_active-1, c_active)==0 && branch_cuts(r_active-1, c_active)==0
137                        adjoin(r_active-1, c_active)=1;
138                    end
139                end
140            end
141        end
142        %figure; imagesc(adjoin), colormap(gray), axis square, axis off, title('Adjoining pixels');
143        %figure; imagesc(IM_unwrapped), colormap(gray), axis square, axis off, title('Pixels unwrapped');
144    end
145end
146
147%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
148% Finally, fill in the branch cut pixels that adjoin the unwrapped pixels.
149% This can be done because the branch cuts actually lie between the pixels,
150% and not on top of them.
151%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
152% disp('Filling in branch cuts that border on unwrapped pixels ...');
153adjoin=zeros(height, width);                                 
154%Re-load the adjoining pixel matrix with the branch cut values:
155for i=2:height-1
156    for j=2:width-1
157       if branch_cuts(i,j)==1 && ...                         %Identify which branch cut pixel borders an unwrapped pixel
158          ( (branch_cuts(i+1,j)==0 || branch_cuts(i-1,j)==0 || branch_cuts(i,j-1)==0 || branch_cuts(i,j+1)==0) ) 
159         adjoin(i,j)=1;
160       end
161    end
162end
163
164[r_adjoin, c_adjoin]=find(adjoin);                          %Obtain coordinates of adjoining unwrapped phase pixels
165
166% Matt changed from 'size' to length
167temp=length(r_adjoin);
168
169for i=1:temp(1)
170    r_active=r_adjoin(i);
171    c_active=c_adjoin(i);
172        %First search below for an adjoining unwrapped phase pixel
173        if unwrapped_binary(r_active+1, c_active)==1
174            phase_ref=IM_unwrapped(r_active+1, c_active);                                   %Obtain the reference unwrapped phase
175            p=unwrap([phase_ref IM_phase(r_active, c_active)]);                             %Unwrap the active pixel
176            IM_unwrapped(r_active, c_active)=p(2);
177            unwrapped_binary(r_active, c_active)=1;                                         %Mark the pixel as unwrapped
178            adjoin(r_active, c_active)=0;                                                   %Remove it from the list of adjoining pixels
179        end
180        %Then search above
181        if unwrapped_binary(r_active-1, c_active)==1
182            phase_ref=IM_unwrapped(r_active-1, c_active);                                   %Obtain the reference unwrapped phase
183            p=unwrap([phase_ref IM_phase(r_active, c_active)]);                             %Unwrap the active pixel
184            IM_unwrapped(r_active, c_active)=p(2);
185            unwrapped_binary(r_active, c_active)=1;                                         %Mark the pixel as unwrapped
186            adjoin(r_active, c_active)=0;                                                   %Remove it from the list of adjoining pixels
187        end
188        %Then search on the right
189        if unwrapped_binary(r_active, c_active+1)==1
190            phase_ref=IM_unwrapped(r_active, c_active+1);                                   %Obtain the reference unwrapped phase
191            p=unwrap([phase_ref IM_phase(r_active, c_active)]);                             %Unwrap the active pixel
192            IM_unwrapped(r_active, c_active)=p(2);
193            unwrapped_binary(r_active, c_active)=1;                                         %Mark the pixel as unwrapped
194            adjoin(r_active, c_active)=0;                                                   %Remove it from the list of adjoining pixels
195        end
196        %Finally search on the left
197        if unwrapped_binary(r_active, c_active-1)==1
198            phase_ref=IM_unwrapped(r_active, c_active-1);                                   %Obtain the reference unwrapped phase
199            p=unwrap([phase_ref IM_phase(r_active, c_active)]);                             %Unwrap the active pixel
200            IM_unwrapped(r_active, c_active)=p(2);
201            unwrapped_binary(r_active, c_active)=1;                                         %Mark the pixel as unwrapped
202            adjoin(r_active, c_active)=0;                                                   %Remove it from the list of adjoining pixels
203        end
204end
205
206end
