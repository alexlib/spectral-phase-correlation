1function [branch_cut_matrix, FLAGS_MATRIX] = calculate_branch_cuts_goldstein(FLAGS_MATRIX, MAX_BOX_SIZE);
  2
  3% Create a matrix containing the residue flags
  4% This matrix contains [height, width] 8-bit values
  5% where the first seven bits of each byte refer in
  6% little-endian order to the values of the following
  7% flags at each pixel: 
  8% 2^0 bit = 1 if positive residue, 0 if not positive residue.
  9% 2^1 bit = 1 if negative residue, 0 if not negative residue.
 10% 2^2 bit = 1 if the pixel lies on a branch cut, 0 if not.
 11% 2^3 bit = 1 if the pixel lies on the image border (I think??), 0 if not.
 12% 2^4 bit = 1 if the pixel represents a balanced residue, 0 if not.
 13% 2^5 bit = 1 if the pixel represnts an "active residue," i.e.
 14%   if the pixel is "connected to the current set of branch cuts."
 15%   I think this means that the pixel is on a branch cut within
 16%   an active search window.
 17% 2^6 bit = 1 if the pixel has been unwrapped.
 18
 19% Define values for the different flags...
 20% Note that matlab indexes bit positions the same way
 21% it does arrays, e.g., little-endian bit 1 corresponds to 2^0.
 22% Positive residue flag bit-position and byte-value-if-true
 23positive_residue_bit_position = 1;
 24
 25% Negative residue flag bit position and byte-value-if-true
 26negative_residue_bit_position = 2;
 27
 28% Branch cut flag bit position and byte-value-if-true
 29branch_cut_bit_position = 3;
 30
 31% Image border flag bit position and byte-value-if-true
 32image_border_bit_position = 4;
 33
 34% Balanced residue flag bit position and byte-value-if-true
 35balanced_charge_bit_position = 5;
 36
 37% Active residue flag bit position and byte-value-if-true
 38active_residue_bit_position = 6;
 39
 40% Unwrapped pixel flag bit position and byte-value-if-true
 41unwrapped_pixel_bit_position = 7;
 42
 43% Flag specifying whether a search box has been centered on the
 44% pixel.
 45been_searched_bit_position = 8;
 46
 47% Initial box size
 48initial_box_size = 3;
 49
 50% Determine matrix height and width
 51[height, width] = size(FLAGS_MATRIX);
 52
 53% Add any existing branch cuts to the branch cut matrix
 54branch_cut_matrix = bitget(FLAGS_MATRIX, branch_cut_bit_position);
 55 
 56% Determine the locations of the residues
 57residue_locs = find(bitget(FLAGS_MATRIX, positive_residue_bit_position) ...
 58    | bitget(FLAGS_MATRIX, negative_residue_bit_position));
 59
 60% Number of residues identified
 61num_residues = length(residue_locs);
 62
 63% Loop over the residue locations.
 64for k = 1 : num_residues
 65   
 66    % Determine the row and column position of the residue.
 67    % The following lines are equivalent to (but faster than):
 68    % [r, c] = ind2sub([height, width], residue_locs(k));
 69    r = rem(residue_locs(k) - 1, height) + 1;
 70    c = (residue_locs(k) - r) / height + 1;
 71    
 72    % Check if the pixel is already balanced.
 73    isBalanced = bitget(FLAGS_MATRIX(r, c), balanced_charge_bit_position);
 74
 75    % Ignore the pixel if it's already balanced.
 76    % or conversely, only continue if the charge isn't already balanced.
 77    if ~isBalanced
 78    
 79        % Mark the residue as "active."
 80        FLAGS_MATRIX(r, c) = bitset(FLAGS_MATRIX(r, c), active_residue_bit_position, 1);
 81        
 82        % Set to balanced
 83        FLAGS_MATRIX(r, c) = bitset(FLAGS_MATRIX(r, c), balanced_charge_bit_position, 1);
 84        
 85        % Calculate the residue charge.
 86        if bitget(FLAGS_MATRIX(r, c), positive_residue_bit_position)
 87            net_charge =  1;
 88        else
 89            net_charge = -1;
 90        end
 91
 92        % Loop over each box size.
 93        for box_size =  initial_box_size : 2 : MAX_BOX_SIZE
 94                      
 95            % Determine the extents of the search box.
 96            [box_rows_01, box_cols_01] = find_box_coordinates([r, c], ...
 97                [height, width], box_size);
 98            
 99            % Count the number of residues in the box
100            num_residues_in_box = count_residues( ...
101            FLAGS_MATRIX(min(box_rows_01) : max(box_rows_01), ...
102                         min(box_cols_01) : max(box_cols_01)));
103             
104             % Find the coordinates of all the residues in the box.
105             [residue_box_rows, residue_box_cols] = ...
106                 find_residue_positions(FLAGS_MATRIX, ...
107                 box_rows_01, box_cols_01, residue_locs(k), ...
108                    num_residues_in_box);
109        
110            % Determine the positions of all of the residues in the box. 
111            % The "_local" name here is meant to indicate that these
112            % are the coordinates in the frame of the box, not in the
113            % overall residue matrix.       
114                                  
115            % Loop over the number of residues contained in the box.
116            for n = 1 : num_residues_in_box
117                
118                % Set the anchor pixel to the n'th residue detected in the
119                % box
120                row_anchor = residue_box_rows(n);
121                col_anchor = residue_box_cols(n);
122                
123                % Determine the extents of the search box centered
124                [box_rows, box_cols] = ...
125                    find_box_coordinates([row_anchor, col_anchor], ...
126                    [height, width], box_size);
127
128                % Determine the number of box pixels.
129                num_box_pixels = length(box_rows);
130            
131                % Loop over the box pixels.
132                for p = 1 : num_box_pixels
133                    
134                    % Get the flags for the box pixel
135                    flag_vals = FLAGS_MATRIX(box_rows(p), box_cols(p));
136                    
137                        % Check if box pixel is a border pixel.
138                        if bitget(flag_vals, image_border_bit_position);
139
140                            % Set the residue to balanced
141                            % if the box contains a border pixel.
142                            FLAGS_MATRIX(row_anchor, col_anchor) = ...
143                                bitset(FLAGS_MATRIX(row_anchor, col_anchor), ...
144                                balanced_charge_bit_position, 1);
145
146                            % Set the net charge in the box to zero
147                            % if the box contains a border pixel.
148                            net_charge = 0;
149
150                            % Place a branch cut between the active pixel
151                            % and the box (border) pixel
152                            [branch_cut_matrix, FLAGS_MATRIX] = ...
153                                place_branch_cut(branch_cut_matrix, ...
154                                FLAGS_MATRIX, [box_rows(p), box_cols(p)],...
155                                [row_anchor, col_anchor]); 
156
157                        % Check if the box pixel is both a residue and not already active   
158                        elseif (bitget(flag_vals, positive_residue_bit_position) ...
159                                || bitget(flag_vals, negative_residue_bit_position));
160
161                            % Check if the pixel is balanced.
162                            isBalanced = ...
163                                bitget(FLAGS_MATRIX(box_rows(p), box_cols(p)), ...
164                                balanced_charge_bit_position);
165
166                            % If the pixel is not already balanced, add its
167                            % polarity to to the net charge and mark as balanced.
168                            if ~isBalanced
169
170                                % Add the pixel's charge to the net charge.
171                                net_charge = net_charge + ...
172                                    get_flags_charge(FLAGS_MATRIX( ...
173                                    box_rows(p), box_cols(p)));
174                               
175                                % Set the pixel to balanced.
176                                FLAGS_MATRIX(box_rows(p), box_cols(p)) = ...
177                                    bitset(FLAGS_MATRIX(box_rows(p), ...
178                                    box_cols(p)), ...
179                                    balanced_charge_bit_position, 1); 
180                            end % End checking for balanced charge.
181
182                            % Set the pixel to active.
183                            FLAGS_MATRIX(box_rows(p), box_cols(p)) = ...
184                               bitset(FLAGS_MATRIX(box_rows(p), ...
185                               box_cols(p)), ...
186                                active_residue_bit_position, 1);
187
188                            % Place a branch cut between that pixel 
189                            % and the residue at which the box is centered.
190                            [branch_cut_matrix, FLAGS_MATRIX] = ...
191                                place_branch_cut(branch_cut_matrix, ...
192                                FLAGS_MATRIX, ...
193                                [row_anchor, col_anchor],...
194                                [box_rows(p), box_cols(p)]);
195
196                        end % End box pixel conditions
197
198                        % Break the loop-over-box-pixels loop
199                        % if the net charge is zero.
200                        if net_charge == 0
201                            break;
202                        end
203                        
204                end % End looping over box pixels.
205                
206                % Break the loop if the net charge goes to zero.
207                if net_charge == 0
208                    break
209                end
210            end
211            
212            % Break the loop if the net charge is zero.
213            if net_charge == 0
214                break
215            end
216                        
217        end % End looping over box size.
218        
219        % If the max box size is reached, place a branch cut from the 
220        % active pixel to the nearest border.
221        if net_charge ~= 0
222              
223            % branch_cut_to_edge
224            [branch_cut_matrix, FLAGS_MATRIX] = ...
225                branch_cut_to_edge(branch_cut_matrix, FLAGS_MATRIX, ...
226                [r, c]);
227            
228        end
229    end  
230    
231end
232
233% Set branch cut pixels to one.
234branch_cut_matrix(branch_cut_matrix > 0) = 1;
235
236end
