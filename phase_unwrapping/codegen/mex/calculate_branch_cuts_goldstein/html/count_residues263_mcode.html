1function NUM_RESIDUES = count_residues(FLAGS_MATRIX)
 2% This function counts the total number of residues contained
 3% in a matrix of residue flags, and also returns the net charge in the
 4% region.
 5
 6% Bit positions of positive and negative residue flags.
 7positive_residue_bit_position = 1;
 8negative_residue_bit_position = 2;
 9
10% NUM_RESIDUES = sum(abs(RESIDUE_MATRIX(:)) > 0);
11NUM_RESIDUES = sum(bitget(FLAGS_MATRIX(:), ...
12    positive_residue_bit_position)) + sum(bitget(FLAGS_MATRIX(:), ...
13    negative_residue_bit_position));
14
15end
